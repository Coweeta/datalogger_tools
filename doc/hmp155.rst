open 5
commands

send 5

smode poll
form 5.2 rh t tw tdf #r#n
form 5.2 rh t tw tdf #t cs4 #r#n
unit m





void set_up_interface(uint8_t address)
{
    rs485_write("form 5.1 rh t #t cs4 #r#n");
    rs485_write("unit m");
    rs485_write("smode poll");
    rs485_write("form 5.1 rh t #t cs4 #r#n");
    sprintf(str, "addr %3d", address) ;
    rs485_write(str);
}
